 41     exercise(app){
 42         app.post('/exercise', (request, response) => {
 43             //creo un nuovo esercizio
 44             var Exercise = require('./model/exercise.js');
 45             var objExercise = new Exercise(request.body.sentence);
 46             //inserisco la frase del nuovo esercizio nel db
 47             objExercise.setKey(this.db.writeSentence(objExercise.getSentence()));
 48             //invia frase ad hunpos per corregerla
 49             var objSolution = this.ad.getHunposSolution(objExercise.getSentence());
 50             const util = require('util');
 51             console.log(util.inspect(objSolution,{depth:null}));
 52             //dalla risposta di hunpos calcola i tags, ovvero la soluzione dell'esercizio
 53             var tags = objExercise.extractTags(objSolution);
 54             //traforma i tags in italiano
 55             var hunposIta = this.convertToIta(tags);
 56 
 57             response.send(this.v.getExercise(objExercise.getSentence(),objExercise.getKey(),hunposIta));
 58         });
 59     }

 61     saveExercise(app){
 62         app.post('/saveExercise', (request, response) => {
 63 
 64             var wordsnumber = request.body.wordsnumber;
 65             var sentence = request.body.sentence;
 66             var key = request.body.key;
 67             var tagsCorrection = this.convertToTags(wordsnumber,request.body);
 68 
 69 
 70             //RIMANE DA SCRIVERE L'ARRAY TAGS RELAZIONATO ALL'ARRAY WORDS NEL DB
 71             this.db.writeSolution(sentence.split(" "), tagsCorrection, sentence, key);
 72             response.send("esercizio inviato");
 73         });
 74     }

 76     convertToTags(wordsnumber, dataCorrection){
 77         var j=0, c=0;//j contatore per l'indice delle tendine, c contatore parola della frase
 78         var tagsCorrection = [];
 79         tagsCorrection.length = wordsnumber;
 80         var actualTag="";
 81         for(var i in dataCorrection) {
 82             //perchè in dataCorrection c'è anche sentence e wordsnumber ma a me servono solo le chiavi per i tags
 83             if(i !== 'sentence' && i !== 'wordsnumber' && i!=='key'){
 84                 if (dataCorrection[i] !== '-') {//se è stato settato qualcosa
 85                     //questi tag non esistono o devono essere settati sulla seconda tendina
 86                     if(dataCorrection[i]==='A' || (dataCorrection[i]==='B' && i===('general'+c)) || (dataCorrection[i]==='E' && i===('general'+c)) || (dataCorrection[i]==='S' && i===('general'+c)) || (dataCorrection[i]==='V' && i===('general'+c))) {
 87                         actualTag += "";
 88                     }
 89                     else{
 90                         actualTag += dataCorrection[i];
 91                     }
 92                 }
 93 
 94                 j++;
 95                 if (j == 14) {
 96                     j = 0;
 97                     tagsCorrection[c]= actualTag;
 98                     c++
 99                     actualTag = "";
100                 }
101             }
102         }
103         return tagsCorrection;
104     }

106     convertToIta(tags){
107         var hunposIta = [];
108         for(var i=0;i<tags.length;i++){
109             hunposIta[i]=this.convertTagToIta(tags[i]);
110         }
111         return hunposIta;
112     }
113 
114     convertTagToIta(tag){
115         var content = this.fs.readFileSync("./js/vocabolario2.json");
116         var jsonContent = JSON.parse(content);
117 
118         var lowercase=tag.split(/[A-Z]{1,2}/);
119         var uppercase=tag.split(/[a-z0-9]+/);
120         var result="";
121         console.log("uppercase[0]: "+uppercase[0]);
122         if(uppercase[0]!='V' && uppercase[0]!='PE' && uppercase[0]!='PC'){
123             for(var i in jsonContent){
124                 if(i==uppercase[0]){
125                     result+=jsonContent[i];
126                 }
127                 if(lowercase[1]){
128                     if(i==lowercase[1]){
129                         result+=" ";
130                         result+=jsonContent[i];
131                     }
132                 }
133             }
134             return result;
135 
136         }
137         for(var x in jsonContent){
138             if(x===tag){
139                 result+=jsonContent[x];
140             }
141         }
142         return result;
143     }
